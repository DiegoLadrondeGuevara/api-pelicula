fields @timestamp, tipo, log_datos.mensaje
| sort @timestamp asc

filter tipo = "INFO"
| fields @timestamp,
         log_datos.pelicula.pelicula_datos.nombre,
         log_datos.pelicula.pelicula_datos.genero,
         log_datos.pelicula.pelicula_datos.fecha,
         log_datos.pelicula.pelicula_datos.duracion_minutos
| sort @timestamp asc

filter tipo = "INFO"
| fields log_datos.pelicula.pelicula_datos.nombre,
         log_datos.pelicula.pelicula_datos.duracion_minutos
| sort log_datos.pelicula.pelicula_datos.duracion_minutos desc

filter tipo = "INFO"
  and log_datos.pelicula.pelicula_datos.genero = "Terror"
| fields @timestamp, log_datos.pelicula.pelicula_datos.nombre

filter tipo = "INFO"
| stats count(*) by log_datos.pelicula.pelicula_datos.genero

filter tipo = "ERROR"
| fields @timestamp, log_datos.mensaje
